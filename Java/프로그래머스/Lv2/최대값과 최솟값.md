class Solution {
    public String solution(String s) {
        String[] arr = s.split(" ");
        
        int min = Integer.parseInt(arr[0]);
        int max = Integer.parseInt(arr[0]);
        
        for (String str : arr) {
            int num = Integer.parseInt(str);
            
            if ( num < min ) min = num;
            if ( num > max ) max = num;
        }
        
        return min + " " + max;
    }
    
}
