import java.util.*;

class Solution {
    public int solution(int[] order) {
        Deque<Integer> stack = new ArrayDeque<>();
        int idx = 0;
        int n = order.length;
        
        for (int box = 1; box <= n; box++) {
            stack.push(box);
            
            while(!stack.isEmpty() && stack.peek() == order[idx]) {
                stack.pop();
                idx++;
                if (idx == n ) break;
            }
        }
        return idx;
    }
}
