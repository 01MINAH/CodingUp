import java.util.*;

class Solution {
    public int[] solution(int n, int s) {
        
        if(n>s) {
            return new int[]{-1};
        }
        int[] answer = new int[n];
        
        int value = s / n;
        int remain = s % n;
        
        Arrays.fill(answer, value);
        
        for(int i = n-1; remain > 0; i--, remain--) {
            answer[i]++;
        }

        return answer;
    }
}
