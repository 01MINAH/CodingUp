import java.util.*;

class Solution {
    public int solution(int[] priorities, int location) {

        Queue<int[]> queue = new LinkedList<>();

        for (int i = 0; i < priorities.length; i++) {
            queue.offer(new int[]{i, priorities[i]});
        }

        int answer = 0;

        while (!queue.isEmpty()) {

            int[] current = queue.poll();

            int max = 0;
            for (int[] q : queue) {
                if (q[1] > max) {
                    max = q[1];
                }
            }

            if (current[1] < max) {
                queue.offer(current);
            } else {
                answer++;

                if (current[0] == location) {
                    return answer;
                }
            }
        }

        return answer;
    }
}
