class Solution {
    public String solution(int[] numbers, String hand) {
        String answer = "";
        int target = 0; // 눌러야할 키패드
        int[] lefts = new int[]{1, 4, 7};
        int[] rights = new int[]{3, 6, 9};
        int left = 10; // 현재 위치 '*'
        int right = 12; // 현재 위치 '#'
        
        for(int i = 0; i < numbers.length; i++){
            target = numbers[i];
            
            if(target == 0) target = 11;
            
            if(isLeft(target, lefts)){
                answer += 'L';
                left = target;
            } else if(isRight(target, rights)){
                answer += 'R';
                right = target;
            } else { // 거리 계산
                int ifL; int ifR;
                ifL = Math.abs(left - target) / 3 + Math.abs(left - target) % 3;
                ifR = Math.abs(right - target) / 3 + Math.abs(right - target) % 3;
                
                if(ifR < ifL) {
                    answer += 'R';
                    right = target;
                } else if(ifR > ifL) {
                    answer += 'L';
                    left = target;
                } else { // 거리 같을 때
                    if(hand.equals("right")) {
                        answer += 'R';
                        right = target;
                    } else {
                        answer += 'L';
                        left = target;
                    }
                    
                }
            }
        }
        return answer;
    }
    
    private boolean isLeft(int target, int[] lefts){
        
        for(int i = 0; i < lefts.length; i++) {
            if(target == lefts[i]){
                return true;
            }
        }
        return false;
    }
    
    private boolean isRight(int target, int[] rights){

        for(int i = 0; i < rights.length; i++) {
            if(target == rights[i]){
                return true;
            }
            
        }
        return false;
    }
    
}
