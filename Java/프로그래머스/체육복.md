### 탐욕 기법 / 그리디 알고리즘

```java
import java.util.*;

class Solution {
    public int solution(int n, int[] lost, int[] reserve) {
        int answer = 0;
        
        int[] clothes = new int[n+2]; // 배열 범위 에러 방지 & 1부터 시작하기 위함
        
        Arrays.fill(clothes, 1);
        
        for(int l : lost) {
            clothes[l]--;
        }
        
        for(int r : reserve){
            clothes[r]++;
        }
        
        // 빌려주기
        for(int i = 1; i<=n; i++){
            if(clothes[i]==0) {
                if(clothes[i-1] == 2) {
                    clothes[i-1]--;
                    clothes[i]++;
                }
                else if(clothes[i+1] == 2) {
                    clothes[i+1]--;
                    clothes[i]++;
                }
            }
        }
        
        for(int i = 1; i<=n; i++) {
            if(clothes[i]>=1){
                answer++;
            }
        }
        
        return answer;
    }
}
```
