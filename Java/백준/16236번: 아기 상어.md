`# bfs`

```java
package Baekjoon;

import java.io.*;
import java.util.*;

public class 아기상어16236 {
    static int n;
    static int[][] N;
    static int sharkX, sharkY, size = 2, eat = 0, time = 0; // 아기상어 위치, 크기, 먹은 크기, 걸린 시간

    static int[] dx = { -1, 0, 1, 0};
    static int[] dy = { 0, -1, 0, 1};

    public static void main(String[] args) throws Exception{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        n = Integer.parseInt(br.readLine());
        N = new int[n][n];

        // N 배열 채우기
        for ( int i = 0; i < N.length; i++ ) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            for (int j = 0; j < N.length; j++) {
                N[i][j] = Integer.parseInt(st.nextToken());
                if ( N[i][j] == 9) {
                    sharkX = i; sharkY = j;
                    N[i][j] = 0;
                }
            }
        }

        while (true) {
            int[] fish = bfs();
            if (fish == null) break;

            int fx = fish[0];
            int fy = fish[1];
            int dist = fish[2];

            // 이동 & 먹기
            sharkX = fx;
            sharkY = fy;
            time += dist;
            eat++;
            N[fx][fy] = 0;

            // 상어 크기 증가
            if ( eat == size ) {
                size++;
                eat = 0;
            }
        }

        System.out.println(time);
    }

    static int[] bfs() {
        boolean[][] visited = new boolean[n][n];
        Queue<int[]> q = new LinkedList<>();
        q.add(new int[] {sharkX, sharkY, 0});
        visited[sharkX][sharkY] = true;

        List<int[]> fishes = new ArrayList<>();
        int minDist = Integer.MAX_VALUE;

        while(!q.isEmpty()) {
            int[] cur = q.poll();
            int x = cur[0];
            int y = cur[1];
            int d = cur[2];

            if (d > minDist) break;

            for (int i = 0; i < 4; i++) {
                int nx = x + dx[i];
                int ny = y + dy[i];

                // 🟡 범위 체크
                if (nx < 0 || ny < 0 || nx >= n || ny >= n) continue;

                // 🔵 방문했거나 상어보다 크면 통과 못함
                if (visited[nx][ny] || N[nx][ny] > size) continue;

                visited[nx][ny] = true;

                // 🐟 먹을 수 있는 물고기면 리스트에 추가
                if (N[nx][ny] != 0 && N[nx][ny] < size) {
                    fishes.add(new int[]{nx, ny, d + 1});
                    minDist = d + 1;
                }

                q.add(new int[] {nx, ny, d + 1});
            }
        }

        if (fishes.isEmpty()) return null;

        // 🍙 정렬: 위 → 왼쪽 순
        fishes.sort((a, b) -> {
            if (a[0] == b[0]) return a[1] - b[1];
            return a[0] - b[0];
        });

        return fishes.get(0);
    }
}
```
