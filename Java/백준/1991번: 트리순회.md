package Baekjoon;

import java.io.*;
import java.util.*;

public class 트리순회1991 {
    static int total;
    static char[] left;
    static char[] right;


    public static void main(String[] args) throws Exception{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        total = Integer.parseInt(br.readLine());
        left = new char[26];
        right = new char[26];

        // 트리 생성
        for (int i = 0; i < total; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());

            char parent = st.nextToken().charAt(0);
            char l = st.nextToken().charAt(0);
            char r = st.nextToken().charAt(0);

            int index = parent - 'A';
            left[index] = l;
            right[index] = r;
        }

        preOrder('A');
        System.out.println();

        inOrder('A');
        System.out.println();

        postOrder('A');
        System.out.println();
    }

    static void preOrder(char a){ // root - l - r


        if (a == '.') return; // gpt 도움 받음!!!!!!!!!

        System.out.print(a);
        preOrder(left[a - 'A']);
        preOrder(right[a - 'A']);

    }

    static void inOrder(char a){ // l - root - r

        if (a == '.') return;

        inOrder(left[a - 'A']);
        System.out.print(a);
        inOrder(right[a - 'A']);
    }

    static void postOrder(char a){ // l - r - root

        if (a == '.') return;

        postOrder(left[a - 'A']);
        postOrder(right[a - 'A']);
        System.out.print(a);
    }
}
