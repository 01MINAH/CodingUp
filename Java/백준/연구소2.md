    static ArrayList<int[]> selected = new ArrayList<>();

    static int[] dx = {-1, 1, 0, 0};
    static int[] dy = {0, 0, -1, 1};

    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());

        N = Integer.parseInt(st.nextToken());
        M = Integer.parseInt(st.nextToken());
        map = new int[N][N];

        for (int i = 0; i < N; i++) {
            st = new StringTokenizer(br.readLine());
            for (int j = 0; j < N; j++) {
                map[i][j] = Integer.parseInt(st.nextToken());

                if (map[i][j] == 0) emptyCnt++;
                else if (map[i][j] == 2) {
                    virusList.add(new int[]{i, j});
                }
            }
        }

        dfs(0, 0);

        System.out.println(answer == Integer.MAX_VALUE ? -1 : answer);
    }

    static void dfs(int idx, int depth) {

        if (depth == M) {
            bfs();
            return;
        }

        for (int i = idx; i < virusList.size(); i++) {
            selected.add(virusList.get(i));   // ì„ íƒ
            dfs(i + 1, depth + 1);

    static void bfs() {
        Queue<int[]> q = new LinkedList<>();
        boolean[][] visited = new boolean[N][N];

        for (int[] v : selected) {
            q.offer(new int[]{v[0], v[1], 0});
            visited[v[0]][v[1]] = true;
        }

        int spreadCnt = 0;
        int time = 0;

        while (!q.isEmpty()) {
            int[] cur = q.poll();
            int x = cur[0];
            int y = cur[1];
            int t = cur[2];

            time = t;

            for (int d = 0; d < 4; d++) {
                int nx = x + dx[d];
                int ny = y + dy[d];

                if (nx < 0 || ny < 0 || nx >= N || ny >= N) continue;
                if (visited[nx][ny] || map[nx][ny] == 1) continue;

                visited[nx][ny] = true;
                q.offer(new int[]{nx, ny, t + 1});

                if (map[nx][ny] == 0) {
                    spreadCnt++;
                }
            }
        }

        if (spreadCnt == emptyCnt) {
            answer = Math.min(answer, time);
        }
    }
